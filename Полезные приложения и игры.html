5. JavaScript (для калькуляторов)

Вот код для калькуляторов:

javascript
// Savings Calculator
document.getElementById('savings-calculator').addEventListener('submit', function(e) {
    e.preventDefault();

    const initialAmount = parseFloat(document.getElementById('initial-amount').value);
    const interestRate = parseFloat(document.getElementById('interest-rate').value) / 100;
    const months = parseInt(document.getElementById('months').value);

    const finalAmount = initialAmount * Math.pow((1 + interestRate / 12), months);
    document.getElementById('savings-result').innerText = `Total Savings: $${finalAmount.toFixed(2)}`;
});

// Loan Calculator
document.getElementById('loan-calculator').addEventListener('submit', function(e) {
    e.preventDefault();

    const loanAmount = parseFloat(document.getElementById('loan-amount').value);
    const loanRate = parseFloat(document.getElementById('loan-rate').value) / 100;
    const loanTerm = parseInt(document.getElementById('loan-term').value);

    const monthlyRate = loanRate / 12;
    const numberOfPayments = loanTerm * 12;
    const monthlyPayment = (loanAmount * monthlyRate) / (1 - Math.pow((1 + monthlyRate), -numberOfPayments));

    document.getElementById('loan-result').innerText = `Monthly Payment: $${monthlyPayment.toFixed(2)}`;
});
